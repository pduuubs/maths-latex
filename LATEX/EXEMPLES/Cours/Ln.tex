\section{Premi\`eres propri\'et\'es}
\label{sec:para1}

La fonction $\exp$ est continue et strictement croissante sur
$\R$. Ses limites en $-\infty$ et $+\infty$ sont respectivement $0$ et
$+\infty$. Donc d'après le théorème des valeurs intermédiaires, pour
tout réel strictement positif $y$, l'équation en $x$:
\begin{equation}
  \label{eq:exy}
  \e^x=y
\end{equation}
a une unique solution que l'on note: $\ln(y)$ appelée logarithme
néperien de~$y$. Ainsi $\ln$ est la fonction réciproque de
$\exp$. Elle est définie sur $\into{0}{+\infty}$.
\begin{equation}
  \label{eq:def}
  \forall y\in \into{0}{+\infty},\quad \e^x=y \iff x=\ln(y)
\end{equation}
\exemples $\e^0=1$ donc $0=\ln(1)$. $\e^1=\e$ donc $1=\ln(\e)$ 
\Def{Pour tout réel $x$ strictement positif on définit le logarithme
néperien de~$x$ noté $\ln(x)$ comme étant l'unique réel $a$ vérifiant:
\[ \exp(a)=x.\quad \text{On a donc: }\quad  \boxed{\exp(\ln(x))=x}\]}
\begin{prop}
On a ainsi défini la fonction logarithme néperien sur
  $\into{0}{+\infty}$. $\ln$ est la fonction réciproque de
$\exp$. Elle  vérifie:
\begin{eqnarray}
  \label{eq:recip}
  \forall x \in \into{0}{+\infty},\quad \e^{\ln(x)}=x\\
  \forall x \in \R, \quad \ln(\e^x)=x \label{eq:recip2}
\end{eqnarray}
\end{prop}
\begin{proof}
\eqref{eq:recip} c'est la définition. On prouve la suivante \eqref{eq:recip2}:
Soit un réel $x$, on note $y=\e^x$. Alors par définition
$x=\ln(y)$. En remplaçant $y$ dans cette dernière par $\e^x$ on a le
résultat.
\end{proof}%, pour tout réel $x$: $\quad \ln(\e^x)=x$}

Comme conséquence de la croissance de $\ln$ sur
  $\into{0}{+\infty}$ on a pour tous réels $a$ et $b$ strictement
  positifs:
\[\ln(a)\leq\ln(b)\iff a\leq b \quad\et{}\quad \ln(a)=\ln(b)\iff a= b\]
\end{prop}

\subsection{Relation fonctionnelle}
Soient deux réels $a$ et $b$ strictement positifs.
\begin{enumerate}
\item Simplifier $\e^{\ln(a)}$, $\e^{\ln(b)}$, $\e^{\ln(ab)}$. Ecrire
  le produit  $ab$ de deux manières.
\item En déduire la relation fonctionnelle: $\boxed{\ln(ab)=
    \phantom{\sum \sum \sum \sum }}$
\item L'exponentielle transforme les sommes en produits. Que fait sa
  fonction réciproque?
\end{enumerate}

\begin{enumerate}
\item Par récurrence on peut montrer que pour tout entier naturel $n$
  on a donc: $\boxed{\ln(a^n)=
    \phantom{\sum \sum \sum  }}$
\item En posant $a=\e^\alpha$ (donc $\alpha=\ldots$) prouver cette
  même propriété directement.

\item Etablir une formule pour $\boxed{\ln(\sqrt a)=
    \phantom{\sum \sum \sum  }}$
\end{enumerate}

\item En calculant $\ln(a\times\frac{1}{a})$ de deux manières,

\item En déduire la formule: $\boxed{\ln(\frac{a}{b})=

\end{enumerate}

\subsubsection{Racine énième}
On peut avec cette nouvelle notion résoudre l'eq en $x$ (solve for
$x$): $x^n=a$ où $a\in\mathbb{R}^+_*$ et $n\in\mathbb{N}^*$
\begin{Def}
  Pour $a\in\mathbb{R}^+_*$ et $n\in\mathbb{N}^*$, il existe un unique
  réel $x$ strictement positif tel que $x^n=a$, c'est
  $x=a^{\frac{1}{n}}$ on l'appelle la racine énième (ou $n$\ieme)
  de~$a$. On note aussi: \[a^{\frac{1}{n}}=\sqrt[n]{a}\]
\end{Def}
\begin{proof} Pour $x$ et $a$ sont strictement positifs:
  \[x^n=a\iff n\ln x=\ln a\iff \ln x=\frac{1}{n}\ln a\iff
  x=\left(\e^{\ln a}\right)^{\frac{1}{n}}\iff x=a^\frac{1}{n}\]
\end{proof}
\begin{rem}
  La racine carrée de $a$ apparait donc (enfin) comme
  $a^\frac12$. \'Ecrire avec des radicaux $x^{\frac32}$ et
  $x^{-\frac23}$ où $x\in\mathbb{R}^+_*$
\end{rem}

\subsection{$\log_a(x)$}
\label{sec:loga}

On veut maintenant définir sur $\mathbb{R}^+_*$ la réciproque de $\exp_a$.
\begin{Def}
  $\log_a(x)=\dfrac{\ln x}{\ln a}$
\end{Def}
\begin{rem}
  C'est la réciproque de la fct $\exp_a$. En effet: 
\end{rem}
\[x\in\mathbb{R}^+_*\implies \exp_a\circ
  \log_a(x)=\e^{\frac{\ln x}{\ln a}\ln a} =\e^{\ln x}=x\]
\[x\in\mathbb{R}\implies \log_a\circ\exp_a (x)=\dots=x\]
\begin{Def}
  En particulier, le logarithme décimal, noté log est le logarithme de
  base~10. $\log(x)=\frac{\ln(x)}{\ln 10}$. C'est la réciproque de
  $x\mapsto 10^x$
\end{Def}
