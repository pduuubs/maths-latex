\begin{exercice}
\textbf{Application aux limites}
 \begin{enumerate}

 \item Limites de $\e^x$.
    \begin{enumerate}
    \item Déduire de \eqref{eq:tang} une preuve de la limite de
    $\e^x$ en $+\infty$.

     \item Déduire de \eqref{eq:tang} que pour tout réel $x$ tel que
    $x<1$:
       \begin{equation}
         \label{eq:mino}
\e^x\leq\frac{1}{1-x}
       \end{equation}

     \item En déduire la limite de $\e^x$ en $-\infty$.
    \end{enumerate}

\item Encadrement de $\e$.\label{enc}
    \begin{enumerate}
    \item Déduire de \eqref{eq:tang} que pour tout $n\in\N^*$:
    \[\left( 1+\frac{1}{n}\right)^n\leq \e\]
    \emph{Indication: }$\frac{1}{n}\times n=1$
    \item Déduire de \eqref{eq:mino} que pour tout $n\in\N^*$:
    \[\e\leq\left( 1+\frac{1}{n}\right)^{n+1}\]
    \end{enumerate}
\item Une suite qui converge vers $\e$ \\
    On considère la suite $(u_n)$ définie par: 
    \[ n\in\N^*,\quad u_n=\left( 1+\frac{1}{n}\right)^n \]
    \begin{enumerate}
    \item Démontrer que pour tout entier $n\in\N^*$:
    \[ 0\leq \e- u_n\leq \frac{\e}{n}\]
    \item En déduire que la suite $(u_n)$ converge vers $\e$
    \item Calculer à l'aide de la calculatrice et de la suite $(u_n)$ une
      valeur approchée à $10^{-6}$ près du nombre réel~$\e$.
    \end{enumerate}%
\end{enumerate}%
 
\end{exercice}
