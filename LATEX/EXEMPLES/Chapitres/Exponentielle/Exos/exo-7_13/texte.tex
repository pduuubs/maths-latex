\begin{exercice}
En posant $X=\e^x$ résoudre dans $\R$ les équations suivantes:
\begin{enumerate}
\item \alignetrois{a) $3\e^{2x}-\e^x-2=0$}{b) $\e^{2x}+\e^x-2=0$}{c) $\e^{2x}-\e^{x+1}-\e^x+\e=0$}
\item \alignetrois{a) $\e^{3x}-\e^x=0$}{b) $2\e^{x}+2\e^{-x}+5=0$}{c) $\e^{x}-\e^{-x}=0$}
\end{enumerate}
 
\end{exercice}
