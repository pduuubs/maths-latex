\begin{exercice}
On considère les trois fonctions suivantes définies sur $\R$:
\[ f(x)=\e^{2x} \qquad g(x)=\e^{x^2} \qquad h(x)=\e^{-x} \qquad i(x)=x\e^x \]
\begin{enumerate}
\item \'Etudier (sens de variation et limites en $-\infty$ et  $+\infty$) ces fonctions.
\item Calculer les valeurs exactes: $f(\ln 3)$, $g(-1)$, $h(\ln 2)$, $i(1)$.
\end{enumerate} 
\end{exercice}
