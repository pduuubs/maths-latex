\begin{exercice}
\begin{enumerate}
  \item Prouver que:
\[ \left(\frac{\sqrt2}{2}+i \frac{\sqrt2}{2}\right)^2=i \]
R\'esoudre alors: $z^4=-1$
\item On note: $j=-\dfrac{1}{2}+i \dfrac{\sqrt3}{2}$. Prouver que:
  $1+j+j^2=0$. En d\'eduire que $j^3=1$. D\'evelopper: $(z-1)(z-j)(z-j^2)$ et enfin, r\'esoudre $z^3=1$

\end{enumerate}
  
\end{exercice}
