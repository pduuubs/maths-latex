\begin{exercice}

 \noindent {\bf Partie A}\\
Soit la fonction $g$ définie sur $\R$ par :
\[ g(x)=\text{e}^x+x \]
 \begin{enumerate}
\item Déterminer les limites de $g$ en $+\infty$ et en $-\infty$.
\item \'Etudier les variations de $g$ sur $\R$.

\item \begin{enumerate}
  \item Justifier le nombre de solutions de $g(x)=0$ sur $\R$.
  \item Donner un encadrement d'amplitude $0,01$ de chaque solution  et
le signe de $g$ sur  $\R$.
  \end{enumerate}
\end{enumerate}
{\bf Partie B}\\ 
Soit la fonction $f$ définie sur $\R^+$ par :
\[ f(x)=\ln\left(\text{e}^x+x\right) \]
On appelle $\Cr$ sa représentation graphique dans un repère orthonormal. 

\begin{enumerate}
\item Déterminer la limite de  $f$  en $+\infty$
\item Dresser le tableau de variation de $f$ .
\item Démontrer que pour tout  $x$  de $\R^+$: $f(x)=x+\ln\left(1+\frac{x}{\text{e}^x}\right)$
\item Etudier la position relative de $\Cr$ et de la droite $\Delta$ d'équation:   $y = x$
\item Déterminer les coordonnées du point $I$ de $\Cr$ où la tangente
  $T$ à
  $\Cr$ est parallèle à $\Delta$.
\item Construire $\Cr$ et $\Delta$, placer $I$ et tracer $T$. (\textit{Unités: 4cm})
  
  \end{enumerate}
\end{exercice}
