\begin{exercice}
On considère l'équation $(E)$: $x^3 - 15 x - 4 = 0$ et l'inéquation:
\begin{equation}
  \label{eq:ineqexo21}
   x^3 - 15 x - 4 > 0 
\end{equation}

  \begin{enumerate}

  \item \textbf{Résolution graphique.}

    \begin{enumerate}
    \item Démontrer que l'équation $(E)$ est équivalente à 
l'équation: 
\[x^2-15=\frac{4}{x}\]
\item Tracer dans un même repère, les courbes représentatives des
  fonctions: 
\[x \longmapsto x^2-15 \quad\text{et}\quad x \longmapsto \frac{4}{x}\]
\item Déterminer graphiquement le nombre de solutions 
de l'équation $(E)$. Une des solutions est un nombre entier, quelle
est sa valeur? Encadrer chacune des autres solutions $\alpha$ et~$\beta$ (avec $\alpha < \beta$) par deux entiers consécutifs.
\item Démontrer que l'inéquation \eqref{eq:ineqexo21} s'écrit:
  $\left]0\,;+\infty\right[$:
 \[x^2-15 >\frac{4}{x}\ \text{sur}\ \left]0\,;+\infty\right[
 \quad\text{et}\quad%
x^2-15 < \frac{4}{x}\ \text{sur}\ \left]-\infty\,;0\right[
 \]

\item Résoudre l'inéquation \eqref{eq:ineqexo21}.
    \end{enumerate}

  \item \textbf{Étude d'une fonction.}\\
Soit $f$ la fonction définie sur $\mathbb{R}$ par $f (x) = x^3 - 15 x -
4$, et $\mathscr{C}_f$ est sa courbe représentative dans un repère.
\begin{enumerate}
\item Justifier la continuité de $f$ sur $\mathbb{R}$.
\item Étudier les limites de $f$ en $-\infty$ et~$+\infty$.
\item Déterminer les variations de $f$ et dresser son tableau de variation.
\item Tracer $\mathscr{C}_f$.
\item Démontrer que l'équation $f (x) = 0$ admet exactement trois
  solutions dans~$\mathbb{R}$.
\item L'une des solutions est un nombre entier, donner un encadrement
  d'amplitude $10^{-3}$ de chacune des autres solutions  $\alpha$ et
$\beta$ (avec $\alpha < \beta$)
\item Étudier le signe de la fonction $f$. En déduire l'ensemble des solutions de l'inéquation~\eqref{eq:ineqexo21}.
\end{enumerate}

\item \textbf{Méthode algébrique.}
  \begin{enumerate}
  \item Déterminer les réels $a$, $b$ et $c$ tels que pour tout
    réel~$x$:
\[ x^3 - 15 x - 4 = (x - 4) (ax^2 + bx + c)\]
\item Résoudre l'équation~$(E)$.
\item Résoudre l'inéquation \eqref{eq:ineqexo21}.
  \end{enumerate}
  \end{enumerate}
\end{exercice}
