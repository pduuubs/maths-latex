\begin{exercice}
On considère la fonction $f$ définie sur $\R$ par:
\[\forall x \in \R^*_+, f(x)=\e^{-\frac{1}{x}}\quad \et{} \quad
\forall x \in \R^-, f(x)=0\]
\begin{enumerate}
\item Prouver que $f$ est continue sur $\R$
\item Prouver que $f$ est dérivable sur $\R$ et que sa dérivée $f'$
  est continue sur $\R$

\item Prouver que $f'$ est dérivable sur $\R$ et que sa dérivée $f''$
  est continue sur $\R$
\end{enumerate}
 
\end{exercice}
