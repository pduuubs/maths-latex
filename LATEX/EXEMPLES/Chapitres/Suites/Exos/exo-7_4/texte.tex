\begin{exercice}
\textbf{Suite arithmético-géométrique.}
Soit la suite $(u_n)$ définie par $u_0$ et: 
\[ \forall n\in\N,\ u_{n+1}=f(u_n)\ \text{où }\ f(x)=ax+b\]
Pour chacune des valeurs de $a$ et $b$ données, répondre aux questions ci-dessous:
\begin{multicols}{3}
  \begin{dingautolist}{172}
\item $a=\dfrac12$ et $b=1$.
\item  $a=-\dfrac23$ et $b=5$
\item $a=1,5$ et $b=-1$ %\label{ques}
\end{dingautolist}
\end{multicols}



\begin{enumerate}
\item Représenter la courbe de $f$ dans un repère orthonormal ainsi que $\Dr$: $y=x$
\item Représenter les premiers termes de la suite sur
  l'axe des abscisses pour $u_0=-4$ et $u_0=6$ (on prendra plutôt
  $u_0=1$ puis $u_0=3$ pour la troisième suite.
%~\ref{ques}) . 
Vérifier quelques termes par le calcul.
\item Déterminer $\alpha$ la solution de $f(x)=x$. (Le réel $\alpha$
  est appelé \emph{point fixe} de~$f$.)
\item On pose $v_n=u_n-\alpha$, et $u_0=6$. Prouver que $(v_n)$ est une suite géométrique de raison~$a$.
\item En déduire l'expression de $u_n$ et la limite éventuelle de $(u_n)$.
\end{enumerate}

\end{exercice}
