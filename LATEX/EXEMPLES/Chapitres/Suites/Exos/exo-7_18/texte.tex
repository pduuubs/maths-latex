\begin{exercice}
\noindent
On cherche si il existe deux réels $a$ et $b$ tels que le polynôme $P$
défini par: 
\[P(x) = \frac{1}{3}x^3 + ax^2 + bx\]
vérifie pour tout nombre réel $x$ la relation:
\begin{equation}
  \label{eq:rela_fonk_exo17}
  P(x+1) - P(x) = x^2
\end{equation}

 
\begin{enumerate} 

\item On suppose dans cette question qu'un tel polynôme existe.
  \begin{enumerate}
 
\item Sans déterminer $a$ et $b$, calculer $P(0)$, $P(1)$ et $P(-1)$. 
 
\item En déduire les seules valeurs possibles pour $a$ et $b$. \label{deterab}

  \end{enumerate}
\item Prouver l'existence et l'unicité du polynôme~$P$ vérifiant les conditions voulues. On vérifiera en
  particulier que l'expression obtenue au~\textbf{\ref{deterab}.} pour $P$ vérifie bien~\eqref{eq:rela_fonk_exo17} pour tout réel~$x$.
 
\item Applications :  
 \begin{enumerate} 
 \item Montrer par récurrence que pour tout entier naturel $n$:  $P(n)\in\mathbb{N}$.
 
 \item On pose pour $n\in\N^*$: $\displaystyle S_n = \sum_{k=1}^n
   k^2=1+2^2+3^2+\cdots +n^2.\ $
 Démontrer les égalités : 
 \[ S_n=P(n+1)=\frac{n(n+1)(2n+1)}{6} \]
 
 \end{enumerate}  
\end{enumerate}  

 \end{exercice}
