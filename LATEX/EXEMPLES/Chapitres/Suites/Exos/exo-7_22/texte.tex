\begin{exercice}
  \begin{enumerate}
  \item Montrer par récurrence qu'on définit bien une suite
$\left(u_{n}\right)_{n\in\mathbb{N}}$ par la donnée de $u_0=5$ et
$u_{n+1}=\sqrt{3u_n+1}$ pour tout $n$ dans $\mathbb{N}$.
\item Prouver alors
que cette suite est décroissante.
\item Prouver que: $ \forall n \in\mathbb{N},\ 0\leq u_n\leq 4$
  \end{enumerate}
 
 \end{exercice}
