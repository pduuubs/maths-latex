\begin{exercice}
Soit $f$ une fonction de $\N^*$ dans $\N^*$ telle que:
\begin{list}{$\bullet$}{}
\item $f(2)=2$
\item $f(mn)=f(m)f(n)$ pour tous $m$ et $n$ de $\N^*$
\item $f$ est strictement croissante \emph{i.e.}
  $m$\textless$n\implies f(m)$\textless$f(n)$
\end{list}
Prouver que $f$ est la fonction identité sur $\N^*$ \emph{i.e.}
$f(n)=n$ pour tout $n$ de $\N^*$
\end{exercice}
