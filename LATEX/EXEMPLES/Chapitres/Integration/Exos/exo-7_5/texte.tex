\begin{exercice}
\hfill{\bf Composées}
\begin{enumerate}
\item \alignetrois{$f(x)=x\e^{x^2}$}{$g(x)=\cos(3x+5)$}{$h(x)=(2x+1)^{2007}$}
\item
  \alignetrois{$f(x)=\dfrac{2x+1}{x^2+x+1}$}{$g(x)=\e^{5x-1}$}{$h(x)=x\sin (x^2-1)$}
\item \alignetrois{$f(x)=\dfrac{x+1}{\sqrt{x^2+2}}$}{$g(x)=\cos
      x\sin x$}{$h(x)=\dfrac{\e^x}{\e^x+1}$}
  \item \alignetrois{$f(x)=\dfrac{\sin(\sqrt x)}{\sqrt
        x}$}{$g(x)=\sin^2x\cos x$}{$h(x)=\dfrac{1}{x\ln x}\quad
      \text{pour }  x>1$}
\item \alignetrois{$f(x)=\dfrac{\ln x}{x}$}{$g(x)=\dfrac{1}{1+\e^{-x}}$}{$h(x)=x\e^x$}
\end{enumerate}
 
\end{exercice}
