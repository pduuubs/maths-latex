\begin{exercice}
\begin{enumerate}
\item %
Montrer que pour tout $t \geq 0$, on a : $1-t^2 \leq 1 \leq
1+t^3$.
\item
En déduire que pour tout $t\geq 0$ :
$\displaystyle 1-t \leq \frac{1}{1+t} \leq 1-t+t^2$.
\item
Calculer pour $x\geq 0$ l'intégrale 
$\displaystyle \int_{0}^{x}  \frac{1}{1+t}  \, dt$.
\item
En déduire que pour tout $x\geq 0$ :
$\displaystyle x-\frac{x^2}{2} \leq \ln(1+x) \leq
x-\frac{x^2}{2}+\frac{x^3}{3}$.
\item
En déduire la limite classique\footnote{On peut la retrouver par
  approximation affine. L'encadrement établi est plus précis que
  ladite approximation. Il donne une approximation de $\ln(1+x)$ à l'ordre
2}: $\displaystyle \lim_{x\to 0^+} \frac{\ln(1+x)}{x} $.

\end{enumerate}

\end{exercice}
