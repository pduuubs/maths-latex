\begin{exercice}
Soit $f$ la fonction définie pour $x\neq 1$ par 
$\displaystyle f(x)= \frac{x-2}{(x-1)^2}$.
\begin{enumerate}
\item
Montrer qu'il existe deux réels $a$ et $b$ tels que pour tout
$x\neq 1$, 
\quad $\displaystyle   f(x) = \frac{a}{(x-1)^2} +\frac{b}{(x-1)}$.
\item
Calculer $\displaystyle \int_{-1}^{0}  f(t) \, dt$.

\end{enumerate}
 
\end{exercice}
