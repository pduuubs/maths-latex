\begin{exercice}
\noindent
Calculer les intégrales suivantes :\\

$\displaystyle \int_{0}^{1} t^2 e^{-t} \, dt$
\hfill
\encadre{
$-5\,{e^{-1}}+2$
}

$\displaystyle \int_{0}^{1} t \sqrt{t^2+1} \, dt$
\hfill
\encadre{
$\frac{2 \sqrt {2}-1}{3}$
}

$\displaystyle \int_{1}^{e} \ln(t) \, dt$
\hfill
\encadre{
$1$
}

$\displaystyle \int_{0}^{\frac{\pi}{2}} \sqrt{1+2t} + \sin(t) \,
dt$
\hfill
\encadre{
$\frac{ \left (1+\pi \right )^{3/2}+2}{3}$
}
\end{exercice}
