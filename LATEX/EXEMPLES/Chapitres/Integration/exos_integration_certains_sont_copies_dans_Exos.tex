\begin{exercice} 
%
\noindent
\`A l'aide d'intégration(s) par parties, calculer :\\

$\displaystyle \int_{0}^{\frac{\pi}{2}}  t \sin(t) \,dt$
\hfill
\encadre{
$1$
}


$\displaystyle \int_{0}^{1} (2-t)e^{t} \, dt$
\hfill
\encadre{
$2e-3$
}


$\displaystyle \int_{0}^{\pi}  (2t+3) \sin(t) \, dt$
\hfill
\encadre{
$2 \pi+6$
}


$\displaystyle \int_{1}^{x}  t\ln(t) \, dt$
\hfill
\encadre{
$\frac{1}{2} {x}^{2}  \ln (x) - \frac{1}{4} {x}^{2}$
}

}



$\displaystyle \int_{0}^{1} t^2 e^{3t}  \, dt$
\hfill
\encadre{
${\frac {5}{27}}\,{e^{3}}-{\frac {2}{27}}$
}



$\displaystyle \int_{0}^{\pi}  e^t \cos(t)  \, dt$
\hfill

$-\frac{1}{2} {e^{\pi }} - \frac{1}{2}$

$\displaystyle \int_{0}^{\pi}  e^{-t} \sin(t)  \, dt$

\end{enumerate}
\end{exercice}%%%

\begin{exercice} %11
%
\noindent
\begin{enumerate}
\item 

Montrer que pour tout $t \geq 0$, on a : $1-t^2 \leq 1 \leq
1+t^3$.
\item
En déduire que pour tout $t\geq 0$ :
$\displaystyle 1-t \leq \frac{1}{1+t} \leq 1-t+t^2$.
\item
Calculer pour $x\geq 0$ l'intégrale 
$\displaystyle \int_{0}^{x}  \frac{1}{1+t}  \, dt$.
\item
En déduire que pour tout $x\geq 0$ :
$\displaystyle x-\frac{x^2}{2} \leq \ln(1+x) \leq
x-\frac{x^2}{2}+\frac{x^3}{3}$.
\item
Retrouver $\displaystyle \lim_{x\to 0^+} \frac{\ln(1+x)}{x} $.

\end{enumerate}
 
\end{exercice}

\begin{exercice} 
%
\noindent
Soit $(I_n)$ la suite d'intégrales définie pour $n\in \NN$ 
par $I_n = \displaystyle \int_{0}^{1} x^n \sqrt{1+x}  \, dx$
pour $n\geq 1$ et 
$I_0 = \displaystyle \int_{0}^{1} \sqrt{1+x}  \, dx$.
\begin{enumerate}
\item 
Calculer $I_0$, puis $I_1$ à l'aide d'une intégration par
parties.
\item
\begin{enumerate}
\item
Comparer $x^n$ et $x^{n+1}$ lorsque $0\leq x \leq 1$.
\item
En déduire que $(I_n)$ est décroissante.
\end{enumerate}
\item
\begin{enumerate}
\item
Grâce à un encadrement de $\sqrt{1+x}$, établir que 
$\displaystyle \frac{1}{n+1} \leq I_n \leq \frac{2}{n+1}$.
\item
En déduire $\displaystyle \lim_{n\to +\infty } I_n$.
\end{enumerate}
\item
\begin{enumerate}
\item
Montrer que pour tout $x \in [0;1]$, 
$\displaystyle 0 \leq \sqrt{2}-\sqrt{1+x} \leq \frac{1}{2}
(1-x)$.
\item
En déduire que 
$\displaystyle \frac{2}{n+1}  - \frac{1}{2n^2} \leq I_n \leq \frac{2}{n+1}$;
et en déduire $\displaystyle \lim_{n\to +\infty } n \, I_n$.
\end{enumerate}

\end{enumerate}
%
\end{exercice}

\begin{exercice}

On pose $I_0 = \int_0^{\frac{\pi}{6}}\sin(3x)\  dx$ et, pour tout nombre $n$
entier naturel non nul, $I_n=\int_0^{\frac{\pi}{6}}x^n\sin(3x)\ dx$.
  \begin{enumerate}[1.]
  \item 
    \begin{enumerate}[a)]
    \item Calculer $I_0$.
    \item En utilisant une intégration par parties, calculer $I_1$.
    \end{enumerate}
\item
    \begin{enumerate}[a)]
    \item En effectuant deux intégrations par parties successives,
      prouver que pour $n \geqslant 1$, 
      $$I_{n+2}=\frac{n+2}{9}\times\left(\frac{\pi}{6}\right)^{n+1}-\frac{(n+2)(n+1)}{9}I_n$$
    \item Calculer $I_3$.
    \end{enumerate}
  \item Sans calculer l'intégrale $I_n$ :
\begin{enumerate}[a)]
  \item Prouver que la suite $(I_n)_{n\in\mathbb{N}}$ est monotone.
  \item Prouver que pour tout nombre  $n$ entier naturel non nul, 
    $$0 \leqslant I_n \leqslant \int_0^{\frac{\pi}{6}}x^n\ dx$$
  \item Déterminer $\lim_{n\to+\infty} I_n$.
  \end{enumerate}
\end{enumerate}

%
%
\end{exercice}
