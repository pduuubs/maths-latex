\begin{exercice}
Soit $n\in\N^*$ et $p$ un entier compris entre $0$ et $n$. 
\begin{enumerate}
\item Supposons $n$ pair. Prouver que: $\displaystyle \binom{n}{p}\leq
  \frac{n!}{\left(\left(\frac{n}{2}\right)!\right)^2}${\small \hfill\rotatebox{180}{On pourra considérer $x\mapsto x(n-x)$}}
\item Supposons $n$ impair. Prouver que: $\displaystyle \binom{n}{p}\leq
  \frac{n!}{\left(\frac{n-1}{2}\right)!\left(\frac{n+1}{2}\right)!}$
\end{enumerate}

 
\end{exercice}
